assesment 1:
create a ansible playbook to install the ngnix services on the target machine

---
- name: Install and verify Nginx
  hosts: target
  become: true
  tasks:
    - name: Install Nginx
      apt:
        name: nginx
        state: present
        update_cache: yes

    - name: Check if Nginx is installed
      command: nginx -v
      register: nginx_check
      ignore_errors: true

    - name: Show Nginx version
      debug:
        var: nginx_check.stderr

simple one> after run > check for the version in target>

to print hello world
code:
---
- name: Install Nginx and display "Hello World"
  hosts: target
  become: true
  tasks:
    - name: Install Nginx
      apt:
        name: nginx
        state: present
        update_cache: yes

    - name: Start and enable Nginx
      service:
        name: nginx
        state: started
        enabled: true

    - name: Create custom index.html
      copy:
        content: "<h1>Hello World</h1>"
        dest: /var/www/html/index.html
        owner: www-data
        group: www-data
        mode: '0644'
